using System.Collections;
using System.Text;
using System.IO;

Hashtable MyVal = new Hashtable();

 public string DML_by_pro(string procedureName, Hashtable Parameters)
    {

        {

            try
            {
                cmd.CommandText = procedureName;
                cmd.CommandType = CommandType.StoredProcedure;
                cmd.Connection = GetOraConn();
                //int loopCounter=0;
                ICollection ParamKeys = Parameters.Keys;
                foreach (object key in ParamKeys)
                {

                    cmd.Parameters.Add(new OracleParameter(key.ToString(), Parameters[key.ToString()]));
                }

                cmd.Connection.Open();
                cmd.ExecuteNonQuery();
                cmd.Connection.Close();
                cmd.Parameters.Clear();
                return "done";
            }

            catch (Exception ex)
            {
                cmd.Parameters.Clear();
                cmd.Connection.Close();
                return ex.Message;
            }


        }
